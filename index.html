window.addEventListener('scroll', function() { const scrolled = window.pageYOffset; const heroTitle = document.querySelector('.hero-title'); const heroSubtitle = document.querySelector('.hero-subtitle'); if (heroTitle && scrolled < window.innerHeight) { // Title moves up slower (parallax) heroTitle.style.transform = `translateY(${scrolled * 0.5}px)`; heroTitle.style.opacity = 1 - (scrolled / 500); // Subtitle fades faster if (heroSubtitle) { heroSubtitle.style.opacity = 1 - (scrolled / 400); } } });
